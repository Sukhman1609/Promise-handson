Q3. "Write the promise function to print numbers from 1 to 7, in which 1 should be printed after 1 sec, 2 should be printed after 2 sec, 3 should be printed after 3 sec, and so on. 

            Numbers
            1
            2
            3
            4
            5
            6
            7"

                function printNumbersWithDelay() {
                    let number = 1;
                  
                    function printNumber() {
                      return new Promise((resolve) => {
                        setTimeout(() => {
                          console.log(number);
                          resolve();
                        }, number * 1000);
                      });
                    }
                  
                    function printNextNumber() {
                      if (number <= 7) {
                        printNumber()
                          .then(() => {
                            number++;
                            printNextNumber();
                          });
                      }
                    }
                  
                    printNextNumber();
                  }
                  
                  printNumbersWithDelay();

Q5. Create examples to explain callback function.

function hello(name, callback) {
            console.log('Hi' + ' ' + name);
            callback();
        }
        
        function callMe() {
            console.log('I am callback function');
        }
        
        hello('Buddy', callMe);

This function that is passed as an argument inside of another function is called a callback function.
